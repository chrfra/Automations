@fastify/static SPA serving (from research — MUST register AFTER API routes):
```typescript
// server/src/plugins/static.ts
import fp from 'fastify-plugin'
import type { FastifyPluginAsync } from 'fastify'
import fastifyStatic from '@fastify/static'
import { fileURLToPath } from 'node:url'
import { join, dirname } from 'node:path'

const __dirname = dirname(fileURLToPath(import.meta.url))

const staticPlugin: FastifyPluginAsync = async (fastify) => {
  if (process.env.NODE_ENV !== 'production') return
  await fastify.register(fastifyStatic, {
    root: join(__dirname, '../../../client/dist'),
    prefix: '/',
  })
  fastify.setNotFoundHandler((_req, reply) => {
    reply.sendFile('index.html')
  })
}

export default fp(staticPlugin)
```

    Create `Dockerfile` at repo root using a multi-stage build:

    ```dockerfile
    # Stage 1: Build Vite client
    FROM node:22-alpine AS client-build
    WORKDIR /app
    COPY client/package*.json ./
    RUN npm ci
    COPY client/ ./
    RUN npm run build

    # Stage 2: Build Fastify server
    FROM node:22-alpine AS server-build
    WORKDIR /app
    COPY server/package*.json ./
    RUN npm ci
    COPY server/ ./
    RUN npm run build

    # Stage 3: Runtime
    FROM node:22-alpine AS runtime
    WORKDIR /app

    # Copy server production dependencies
    COPY --from=server-build /app/package*.json ./
    RUN npm ci --omit=dev

    # Copy built server output
    COPY --from=server-build /app/dist ./dist

    # Copy built Vite output (Fastify serves it as static files)
    COPY --from=client-build /app/dist ./client/dist

    # Data directory (DB + key will be volume-mounted)
    RUN mkdir -p data/secrets

    EXPOSE 3000

    CMD ["node", "dist/index.js"]
    ```

    Create `docker-compose.yml` at repo root as shown in the interfaces block above. Important notes:
    - Both `test-secrets` and `prod-secrets` are separate named volumes — this prevents key file bleed between environments

    - Services use `profiles` so `docker compose up` without `--profile` starts nothing (requires explicit `--profile test` or `--profile prod`)
